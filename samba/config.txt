(2) set by config file
=============================
modified /etc/samba/smb.conf

# cat >> /etc/samba/smb.conf << EOF

[workspace]
	comment         = Public Areas
	path            = /workspace
	browseable      = yes
	public          = yes
	writable        = yes
	create mode     = 0775
	directory mode  = 0775
	printable       = no
	guest ok        = yes
	;write list     = +staff

EOF
